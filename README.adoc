== Neo4j Docker server

.start database:
 
 docker run --name neo4j -d --rm --publish=7474:7474 --publish=7687:7687 --volume=$HOME/neo4j/data:/data \--volume=$HOME/neo4j/logs:/logs neo4j:3.3.0

.stop database:

 docker stop neo4j
